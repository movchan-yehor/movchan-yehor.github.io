Основи SQL — частина 2

DISTINCT, AS, ORDER BY, LIMIT та OFFSET

DISTINCT — унікальні значення
SELECT DISTINCT колонка FROM таблиця;
Одна колонка
Повертає тільки унікальні міста — без повторів.
SELECT DISTINCT city FROM users;

Кілька колонок
DISTINCT розглядає комбінацію всіх обраних колонок — унікальними мають бути пари.
SELECT DISTINCT city, status FROM users;
Увага: DISTINCT завжди стоїть одразу після SELECT, до списку колонок. Не можна написати SELECT name, DISTINCT city.
AS — псевдоніми колонок і таблиць
SELECT вираз AS псевдонім FROM таблиця AS псевдонім;
Псевдонім колонки
Змінює назву у результаті — зручно для читабельності.
SELECT
  name  AS full_name,
  age   AS years
FROM users;
Псевдонім таблиці
Скорочує назву таблиці — особливо корисно при JOIN.
SELECT u.name, u.city
FROM users AS u
WHERE u.age > 18;
Псевдонім для виразу або функції
Будь-який обчислений вираз можна назвати зрозуміло.
SELECT
  name,
  price * 1.2        AS price_with_vat,
  UPPER(name)       AS name_upper,
  COUNT(*)          AS total_users
FROM products;
Примітка: слово AS можна опустити — name full_name теж працює. Але з AS читабельніше.
ORDER BY — сортування результату
SELECT ... FROM ... ORDER BY колонка [ASC | DESC];
ASC — за зростанням
За замовчуванням — можна не писати.
SELECT * FROM users
ORDER BY age ASC;
DESC — за спаданням
Від найбільшого до найменшого.
SELECT * FROM users
ORDER BY age DESC;
Сортування за кількома колонками
Спочатку сортує за першою колонкою, при однакових значеннях — за другою.
SELECT name, city, age FROM users
ORDER BY city ASC, age DESC;

Сортування за псевдонімом або номером
Можна посилатись на AS-псевдонім або порядковий номер колонки у SELECT.
SELECT name, age * 2 AS double_age
FROM users
ORDER BY double_age DESC;

-- або за номером: ORDER BY 2 DESC
LIMIT та OFFSET — обмеження кількості рядків
SELECT ... FROM ... ORDER BY ... LIMIT n OFFSET m;
LIMIT — перші N рядків
SELECT * FROM users
ORDER BY age DESC
LIMIT 5;
-- 5 найстарших
OFFSET — пропустити M рядків
SELECT * FROM users
ORDER BY age DESC
LIMIT 5 OFFSET 10;
-- рядки 11–15
Пагінація — посторінкове виведення
Типовий патерн для сторінок: OFFSET = (сторінка − 1) × розмір
-- Сторінка 1 (рядки 1–10)
SELECT * FROM products ORDER BY id LIMIT 10 OFFSET 0;

-- Сторінка 2 (рядки 11–20)
SELECT * FROM products ORDER BY id LIMIT 10 OFFSET 10;

-- Сторінка 3 (рядки 21–30)
SELECT * FROM products ORDER BY id LIMIT 10 OFFSET 20;
Увага: LIMIT без ORDER BY повертає рядки у довільному порядку — результат може бути різним при кожному виконанні. Завжди додавай сортування.
Все разом — повний приклад
Топ-5 унікальних міст з найбільшою кількістю активних користувачів старших 18 років, починаючи з 6-го результату.
SELECT DISTINCT
  city                  AS user_city,
  COUNT(*)              AS total
FROM  users             AS u
WHERE u.age > 18
  AND  u.status = 'active'
ORDER BY total DESC
LIMIT  5
OFFSET 5;
