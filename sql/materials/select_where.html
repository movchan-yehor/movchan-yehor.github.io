Основи SQL

SELECT та WHERE — вибірка та фільтрація даних

SELECT — вибірка даних
SELECT колонки FROM таблиця;
Вибрати всі колонки
Символ * означає «всі поля».
SELECT * FROM users;
Вибрати конкретні колонки
Колонки перераховуються через кому.
SELECT name, email, age
FROM users;
WHERE — фільтрація рядків
SELECT ... FROM таблиця WHERE умова;
Проста умова
SELECT * FROM users
WHERE age = 25;

SELECT * FROM users
WHERE city = 'Київ';
Кілька умов — AND / OR
SELECT * FROM users
WHERE age > 18 AND city = 'Львів';

SELECT * FROM users
WHERE age < 16 OR age > 60;
Оператори у WHERE
=   !=   <>
Рівно / не рівно
WHERE age = 30
WHERE city != 'Одеса'
<   >   <=   >=
Порівняння чисел та дат
WHERE age >= 18
WHERE price < 1000
BETWEEN
Значення у діапазоні (включно)
WHERE age BETWEEN 18 AND 35
IN
Збіг з одним зі списку
WHERE city IN ('Київ', 'Харків')
NOT IN
Немає у списку
WHERE status NOT IN ('banned', 'deleted')
LIKE
Пошук за шаблоном
% — будь-яка кількість символів
_ — рівно один символ
WHERE name LIKE 'Іван%'
WHERE code LIKE 'UA_'
IS NULL
Значення відсутнє
WHERE phone IS NULL
WHERE phone IS NOT NULL
NOT
Заперечення умови
WHERE NOT city = 'Херсон'
WHERE NOT age BETWEEN 20 AND 30
Комбінований приклад
Вибрати ім'я та email активних користувачів з Київа або Харкова, старших 18 років, чий email вказано.
SELECT name, email
FROM   users
WHERE  status = 'active'
  AND  age > 18
  AND  city IN ('Київ', 'Харків')
  AND  email IS NOT NULL;
