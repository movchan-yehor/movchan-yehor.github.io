CardsGrid(tk.Frame) не чіпати · керує сіткою 
class CardsGrid(tk.Frame):
    COLS = 4

    def add_card(self, key: str) → WeatherCard:
        card = WeatherCard(self)
        self.cards[key] = card
        self._reposition()
        return card          # ← повертає картку в App

    def _reposition(self):
        for idx, (_, card) in enumerate(self.cards.items()):
            card.grid(          # ← картка — просто Frame у grid
                row=idx // self.COLS,
                column=idx % self.COLS,
                sticky="nsew"
            )
